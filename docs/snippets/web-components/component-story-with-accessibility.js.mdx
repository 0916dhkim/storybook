// Button.stories.js

import { Button } from './Button';

export default {
  title: 'Accessibility testing',
  render: (args) => Button(args),
  argTypes: {
    backgroundColor: { control: 'color' },
  },

// This is an accessible story

    label: 'Button',
  },

  args: {
