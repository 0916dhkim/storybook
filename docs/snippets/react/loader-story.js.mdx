import fetch from 'node-fetch';

import { TodoItem } from './TodoItem';

  component: TodoItem,
};

    return <TodoItem {...args} {...todo} />;

  loaders: [
    async () => ({
      todo: await (await fetch('https://jsonplaceholder.typicode.com/todos/1')).json(),
    }),
  ],
};
