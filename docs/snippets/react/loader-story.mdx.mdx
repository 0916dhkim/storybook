import { Meta, Story } from '@storybook/addon-docs';

import fetch from 'node-fetch';

import { TodoItem } from './TodoItem';

  name="Primary"
  loaders={[
    async () => ({

      ).json(),
    }),
  ]}
  render={(args, { loaded: { todo } }) => <TodoItem {...args} todo={todo} />} />
