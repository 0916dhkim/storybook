// TodoItem.stories.js

import fetch from 'node-fetch';

import TodoItem from './TodoItem.svelte';

export default {
  component: TodoItem,
};

export const Primary = {
  render: (args, { loaded: { todo } }) => ({
    component: TodoItem,
    props: {
      ...args,
      todo,
    },
  }),
  loaders: [
    async () => ({
      todo: await (await fetch('https://jsonplaceholder.typicode.com/todos/1')).json(),
    }),
  ],
};
