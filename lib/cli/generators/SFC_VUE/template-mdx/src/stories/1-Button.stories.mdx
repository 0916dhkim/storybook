# Button

This `Button` documentation defines three stories:

<Story name="text">
  {() => ({
    components: {
      MyButton,
    },
    template: '<my-button @click="action">Hello Button</my-button>',
    methods: {
      action: action('clicked'),
    },
  })}
</Story>

<Story name="jsx">
  {() => ({
    components: {
      MyButton,
    },
    render() {
      return <my-button onClick={this.action}>With JSX</my-button>;
    },
    methods: {
      action: linkTo('clicked'),
    },
  })}
</Story>

<Story name="emoji">

</Story>
